public class Collision {
public double probability(int[] ass, int ids) {
  // naive = nice
  try {
  double z1 = 1, z2=1;
  int left = ids, vfra1 = ids, vfra2 = ids;
  for(int i=ass.length;i-->0;) {
    vfra1 = ids;
    for(int j=ass[i];j-->0;) {
      z1 = z1 * left / vfra1;
      z2 = z2 * left / vfra2;
      left--;
      vfra1--;
    }
  }
  if(left<0) return 0;
  return z1-z2;
}catch(Exception e) {
  return 0;
}
